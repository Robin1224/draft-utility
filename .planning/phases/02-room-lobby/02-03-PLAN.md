  <action>
    Create `src/live/room.js`:
    1. Import `live`, `LiveError` from `svelte-realtime/server`; `db` from `$lib/server/db`; `topicForRoom`, `loadLobbySnapshot`, `joinTeamForUser`, `upsertGuestSpectator`, `getRoomByPublicCode` from `$lib/server/rooms.js` (adjust imports to match Task 1 exports).
    2. Export `lobby = live.stream(
         (ctx, publicCode) => topicForRoom(publicCode),
         async (ctx, publicCode) => {
           const code = typeof publicCode === 'string' ? publicCode : String(publicCode);
           const room = await getRoomByPublicCode(db, code);
           if (!room) throw new LiveError('NOT_FOUND', 'Room not found');
           if (ctx.user?.role === 'guest' && ctx.user?.guestId) {
             await upsertGuestSpectator(db, room.id, ctx.user.guestId);
           }
           const snap = await loadLobbySnapshot(db, code);
           if (!snap) throw new LiveError('NOT_FOUND', 'Room not found');
           return snap;
         },
         { merge: 'set', access: () => true }
       );`
    3. Export `joinTeam = live(async (ctx, publicCode, team) => {
         if (ctx.user?.role !== 'player' || !ctx.user?.id) {
           throw new LiveError('UNAUTHORIZED', 'Sign in to join a team');
         }
         if (team !== 'A' && team !== 'B') throw new LiveError('VALIDATION', 'Invalid team');
         const room = await getRoomByPublicCode(db, publicCode);
         if (!room) throw new LiveError('NOT_FOUND', 'Room not found');
         if (room.phase !== 'lobby') throw new LiveError('FORBIDDEN', 'Team changes are locked');
         await joinTeamForUser(db, { roomId: room.id, userId: ctx.user.id, team });
         const snap = await loadLobbySnapshot(db, publicCode);
         ctx.publish(topicForRoom(publicCode), 'set', snap);
         return snap;
       });`
    4. Do **not** edit `hooks.ws.js` in this plan (guestId arrives in 02-04). Document that production kick requires 02-04.
